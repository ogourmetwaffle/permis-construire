create table public.documents (

    id bigint generated always as identity primary key,

    dossier_id bigint not null references dossiers(id) on delete cascade,

    nom_fichier text not null,

    chemin_storage text not null,

    taille bigint,

    type_mime text,

    created_at timestamp default now()
);


-- Ajout de la date d'archivage des documents
ALTER TABLE public.documents
ADD COLUMN archived_at TIMESTAMPTZ NULL;

-- Index pour accélérer les recherches sur les documents actifs/archivés
CREATE INDEX IF NOT EXISTS idx_documents_archived_at
ON public.documents (archived_at);
